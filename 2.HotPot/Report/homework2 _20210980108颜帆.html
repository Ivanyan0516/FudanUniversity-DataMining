from statsmodels.formula.api import ols
#
def forward_select(data,target):
    #variate=set(data.columns)  #
    variate = {'','','62','','','', '',''}
    variate.remove(target)  #
    selected=[]
    current_score,best_new_score=float('inf'),float('inf')  #AIC
    #
    while variate:
        aic_with_variate=[]
        for candidate in variate:  #
            formula="{}~{}".format(target,"+".join(selected+[candidate]))  #
            aic=ols(formula=formula,data=data).fit().aic  #olsaic
            aic_with_variate.append((aic,candidate))  #aic
        aic_with_variate.sort(reverse=True)  #aic
        best_new_score,best_candidate=aic_with_variate.pop()  #aic
        if current_score>best_new_score:  #aicaic
            variate.remove(best_candidate)  #
            selected.append(best_candidate)  #
            current_score=best_new_score  #
            print("aic is {},continuing!".format(current_score))  #aic
        else:
            print("for selection over!")
            break
    formula="{}~{}".format(target,"+".join(selected))  #
    print("final formula is {}".format(formula))
    model=ols(formula=formula,data=data).fit()
    return(model)
